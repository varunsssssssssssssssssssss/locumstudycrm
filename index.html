const HEADERS = ['id','name','type','org','jobTitle','stage','channel','email','phone','url','owner','dateContacted','followup','notes','activityLog','added']; function formatPhone(p) { if (!p) return ''; p = String(p).trim(); // Convert +61 3 XXXX XXXX → 03 XXXX XXXX p = p.replace(/^\+61\s*3\s*/, '03 '); // Convert +61 4XX → 04XX p = p.replace(/^\+61\s*4/, '04'); // Convert +61 2 → 02 p = p.replace(/^\+61\s*2\s*/, '02 '); // Remove leading apostrophe from Apollo exports p = p.replace(/^'/, ''); return p; } function styleSheet(sheet, numRows) { const lastCol = HEADERS.length; // Header row styling const headerRange = sheet.getRange(1, 1, 1, lastCol); headerRange.setBackground('#0a0a0a'); headerRange.setFontColor('#ffffff'); headerRange.setFontWeight('bold'); headerRange.setFontSize(10); // Data rows - alternating subtle background if (numRows > 0) { for (let i = 2; i <= numRows + 1; i++) { const bg = i % 2 === 0 ? '#f9f9f9' : '#ffffff'; sheet.getRange(i, 1, 1, lastCol).setBackground(bg); } sheet.getRange(2, 1, numRows, lastCol).setFontSize(10); sheet.getRange(2, 1, numRows, lastCol).setFontColor('#333333'); } // Column widths sheet.setColumnWidth(1, 60); // id sheet.setColumnWidth(2, 160); // name sheet.setColumnWidth(3, 120); // type sheet.setColumnWidth(4, 180); // org sheet.setColumnWidth(5, 200); // jobTitle sheet.setColumnWidth(6, 90); // stage sheet.setColumnWidth(7, 100); // channel sheet.setColumnWidth(8, 200); // email sheet.setColumnWidth(9, 120); // phone sheet.setColumnWidth(10, 220); // url sheet.setColumnWidth(11, 80); // owner sheet.setColumnWidth(12, 110); // dateContacted sheet.setColumnWidth(13, 90); // followup sheet.setColumnWidth(14, 200); // notes sheet.setColumnWidth(15, 200); // activityLog sheet.setColumnWidth(16, 90); // added // Freeze header row sheet.setFrozenRows(1); // Phone column as plain text to prevent formula errors sheet.getRange(1, 9, sheet.getMaxRows(), 1).setNumberFormat('@'); } function doGet(e) { const action = e.parameter.action || 'read'; const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); if (action === 'init') { sheet.clearContents(); sheet.appendRow(HEADERS); styleSheet(sheet, 0); return jsonOut({ok: true}); } const rows = sheet.getDataRange().getValues(); if (rows.length < 2) return jsonOut([]); const headers = rows[0]; const data = rows.slice(1).map(r => { const o = {}; headers.forEach((h, i) => o[h] = r[i]); return o; }); return jsonOut(data); } function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const payload = JSON.parse(e.postData.contents); if (payload.action === 'sync') { sheet.clearContents(); sheet.appendRow(HEADERS); payload.data.forEach(c => { const row = HEADERS.map(h => { if (h === 'phone') return formatPhone(c[h]); return c[h] || ''; }); sheet.appendRow(row); }); styleSheet(sheet, payload.data.length); return jsonOut({ok: true, count: payload.data.length}); } return jsonOut({ok: false, error: 'unknown action'}); } function jsonOut(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
